function repeatWithRules(word) {
    let res = ""
    for (let i = 0; i < word.length; i++) {
        if (i === 0)
            res += word[i]
        else if (i === 1)
            res += word[i].repeat(2)
        else if (i === 2)
            res += word[i].repeat(3)
        else
            res += word[i].repeat(4)

```js
function dashBetweenOdd(num) {
    let str = num.toString();
    let result = "";
    for (let i = 0; i < str.length; i++) {
        result += str[i];
        if (
            i < str.length - 1 &&
            str[i] % 2 !== 0 &&
            str[i + 1] % 2 !== 0
        ) {
            result += "-";
        }
    }
    return result;
}
```
